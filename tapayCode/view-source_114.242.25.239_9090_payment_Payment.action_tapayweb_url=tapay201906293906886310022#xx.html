function getPsdtxt(val) {
//val为密码值
console.log(val);
//使用MD5加密
var md5word = hex_md5(val);
var userid = document.getElementById('userid').value;
var orderNum = document.getElementById('orderNum').value;
var tranNo = document.getElementById('tranNo').value;
//ajax判断成功
$.ajax({
type: "POST",
url: sy.contextPath+"/ajax/payment_OnlionPay.action",
data: "userid="+userid+"&orderNum="+orderNum+"&tranNo="+tranNo+"&payPwd="+md5word,
dataType: 'json',
success: function(result){
mui.showLoading("正在加载..", "div");
if (result.success) {
//关闭加载中
mui.hideLoading();
window.location.href =sy.contextPath+"/payment_successPay.action?orderNum=" +orderNum ;
} else if(result.id=='1' && !result.success){
//ajax判断失败
mui.hideLoading();
mui('#psdbox').popover('toggle');
mui.alert("", "密码错误，请重新输入！", "确定", function () {
mui('#psdbox').popover('toggle');
getPsd();
});
}else if(!result.success){
alert(result.msg);
mui.hideLoading();
/* //关闭加载中
mui.hideLoading();
window.location.href =sy.contextPath+"/payment_successPay.action?orderNum=" +orderNum ; */
}
}
});
}
